    (let [a (glsl-spec nil "(a)")
          b (glsl-spec [a] "(b)")
          c (glsl-spec [a] "(c)")
          d1 (glsl-spec [b] "(d1)")
          d2 (glsl-spec [b c] "(d2)")]
      (is (= "(a)(b)(d1)(x)" (glsl/assemble (glsl-spec [d1] "(x)"))))
      (is (= "(a)(c)(b)(d2)(x)" (glsl/assemble (glsl-spec [d2] "(x)"))))))
